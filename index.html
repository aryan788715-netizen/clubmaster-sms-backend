5Create Edge Function

Create the SMS OTP function:

supabase functions new send-otp

This creates: supabase/functions/send-otp/index.ts

Replace the content with:

import { serve } from "https://deno.land/std@0.168.0/http/server.ts"

const MSG91_AUTH_KEY = Deno.env.get('MSG91_AUTH_KEY')!
const MSG91_TEMPLATE_ID = Deno.env.get('MSG91_TEMPLATE_ID') || 'default'

serve(async (req) => {
  // Handle CORS
  if (req.method === 'OPTIONS') {
    return new Response('ok', {
      headers: {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'POST',
        'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
      }
    })
  }

  try {
    const { phone, otp } = await req.json()

    if (!phone || !otp) {
      throw new Error('Phone and OTP are required')
    }

    // Send SMS via MSG91
    const msg91Response = await fetch('https://api.msg91.com/api/v5/otp', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'authkey': MSG91_AUTH_KEY
      },
      body: JSON.stringify({
        template_id: MSG91_TEMPLATE_ID,
        mobile: phone.replace('+', ''),
        otp: otp,
        otp_expiry: 5 // 5 minutes
      })
    })

    const result = await msg91Response.json()

    if (result.type === 'success') {
      return new Response(
        JSON.stringify({ 
          success: true, 
          message: 'OTP sent successfully',
          phone: phone 
        }),
        {
          headers: {
            'Content-Type': 'application/json',
            'Access-Control-Allow-Origin': '*'
          },
          status: 200
        }
      )
    } else {
      throw new Error(result.message || 'Failed to send SMS')
    }

  } catch (error) {
    return new Response(
      JSON.stringify({ 
        success: false, 
        error: error.message 
      }),
      {
        headers: {
          'Content-Type': 'application/json',
          'Access-Control-Allow-Origin': '*'
        },
        status: 400
      }
    )
  }
})

6Set Environment Variables

Create .env file in your project root:

MSG91_AUTH_KEY=your_msg91_auth_key_here
MSG91_TEMPLATE_ID=your_template_id_here

Set secrets in Supabase:

supabase secrets set MSG91_AUTH_KEY=your_auth_key_here
supabase secrets set MSG91_TEMPLATE_ID=your_template_id_here

7Deploy Edge Function

Link to your Supabase project:

supabase link --project-ref your-project-ref

Deploy the function:

supabase functions deploy send-otp
‚úÖ Success! Your function is now deployed at:
https://your-project-ref.supabase.co/functions/v1/send-otp
